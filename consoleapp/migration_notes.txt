Run from the solution folder

-- Add a new migration
dotnet ef migrations add <MigrationNameNoSpacePascalCase> -s consoleapp\consoleapp.csproj -p <TargetProjectPath>

-- Remove an existing migration
dotnet ef migrations remove <MigrationNameNoSpacePascalCase> -s consoleapp\consoleapp.csproj -p <TargetProjectPath>

-- Script out all migrations
dotnet ef migrations script -s consoleapp\consoleapp.csproj -p <TargetProjectPath> -o <ScriptName>.sql
